#git functions
git status تعرض لك حالة المستودع
git branch تعرض لك الفروع التي عندك
git clone تأخذ نسخ من مستودع خارجي
git add اضافة الملفات الى قائمة الانتظار 
git reset سحب الملفات من قلئمة الانتظار 
git commit اضافة الملفات من قائمة الانتظار الى المستودع
git push نقل ملفات المستودع المحلي الى المستودع الخارجي
git pull جلب الملفات المعدلة من المستودع الخارجي
